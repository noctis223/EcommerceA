function initializeKeycloak(keycloak: KeycloakService) {
  return () =>
    keycloak.init({
      config: {
        url: ADDRESS_AUTHENTICATION_SERVER,
        realm: 'Angular',
        clientId: 'Ecommerce'
      },

      initOptions: {
        onLoad: 'check-sso',

      },
      enableBearerInterceptor : true

    });






}
